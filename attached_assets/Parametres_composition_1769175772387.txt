// Fonctions de distribution pour la g√©n√©ration proc√©durale
const mathUtils = {
  // Distribution normale (Gaussienne)
  gaussianRandom: (mean, std) => {
    let u = 0, v = 0;
    while (u === 0) u = Math.random();
    while (v === 0) v = Math.random();
    return mean + std * Math.sqrt(-2.0 * Math.log(u)) * Math.cos(2.0 * Math.PI * v);
  },
  
  // Distribution b√™ta (pour les probabilit√©s)
  betaRandom: (mean, std) => {
    // Approximation simple pour distribution b√™ta
    const alpha = mean * ((mean * (1 - mean) / (std * std)) - 1);
    const beta = (1 - mean) * ((mean * (1 - mean) / (std * std)) - 1);
    
    // G√©n√©rateur b√™ta simplifi√©
    const x = Math.random();
    const y = Math.random();
    return Math.pow(x, alpha - 1) * Math.pow(1 - x, beta - 1);
  },
  
  // Distribution de Poisson
  poissonRandom: (lambda) => {
    let L = Math.exp(-lambda);
    let k = 0;
    let p = 1;
    
    do {
      k++;
      p *= Math.random();
    } while (p > L);
    
    return k - 1;
  },
  
  // Choix pond√©r√© al√©atoire
  weightedRandomChoice: (weights) => {
    const total = Object.values(weights).reduce((a, b) => a + b, 0);
    let random = Math.random() * total;
    
    for (const [key, weight] of Object.entries(weights)) {
      if (random < weight) return key;
      random -= weight;
    }
  },
  
  // Normalisation lin√©aire
  normalize: (value, min, max) => {
    return Math.min(1, Math.max(0, (value - min) / (max - min)));
  },
  
  // Classification du score Vermeer
  classifyVermeerScore: (score) => {
    if (score >= 0.90) return "VERMEER_PERFECT";
    if (score >= 0.80) return "VERMEER_TYPICAL";
    if (score >= 0.70) return "VERMEER_ATYPICAL";
    if (score >= 0.60) return "INFLUENCE_VERMEER";
    return "NON_VERMEER";
  }
};
